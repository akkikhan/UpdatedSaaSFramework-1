// Install NuGet packages Install-Package SaaSFramework.Authentication Install-Package SaaSFramework.RBAC // Program.cs builder.Services.AddSaaSFramework(options => { options.AuthApiKey = "auth_key_here"; options.RbacApiKey = "rbac_key_here"; options.BaseUrl = "https://api.yourplatform.com"; }); // Usage in Controller [ApiController] public class WeatherController : ControllerBase { private readonly ISaaSAuth _auth; private readonly ISaaSRBAC _rbac; public WeatherController(ISaaSAuth auth, ISaaSRBAC rbac) { _auth = auth; _rbac = rbac; } [HttpGet] public async Task<IActionResult> GetWeather() { var user = await _auth.ValidateToken(Request.Headers.Authorization); var hasPermission = await _rbac.CheckPermission(user.Id, "weather:read"); if (!hasPermission) return Forbid(); return Ok(GetWeatherData()); } }

// Install npm packages npm install @saas-framework/auth @saas-framework/rbac // server.js const { SaaSAuth, SaaSRBAC } = require('@saas-framework/core'); const auth = new SaaSAuth({ apiKey: 'auth_key_here', baseUrl: 'https://api.yourplatform.com' }); const rbac = new SaaSRBAC({ apiKey: 'rbac_key_here', baseUrl: 'https://api.yourplatform.com' }); // Middleware async function authenticateUser(req, res, next) { try { const token = req.headers.authorization?.split(' ')[1]; req.user = await auth.validateToken(token); next(); } catch (error) { res.status(401).json({ error: 'Unauthorized' }); } } async function requirePermission(permission) { return async (req, res, next) => { const hasPermission = await rbac.checkPermission(req.user.id, permission); if (!hasPermission) { return res.status(403).json({ error: 'Forbidden' }); } next(); }; } // Routes app.get('/api/weather', authenticateUser, requirePermission('weather:read'), (req, res) => { res.json({ weather: 'sunny', temperature: 75 }); } );
